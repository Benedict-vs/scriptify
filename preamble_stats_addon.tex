% =============================================================================
%  preamble_stats_addon.tex
%  Included by main.tex (see templates/main.tex), immediately BEFORE
%  \begin{document}. It adds theorem environments + statistics macros. It does
%  NOT redefine anything from the `physics` package (\abs, \norm, \tr, \Tr,
%  \pdv, \dv, ...), so it coexists with an existing preamble unchanged.
%
%  Language: the printed theorem labels below ("Theorem", "Example", ...) are
%  the one language-dependent part of this file. For a script in another
%  language, translate the LABELS and keep the ENVIRONMENT NAMES — the names
%  are code (CLAUDE.md and the fragments reference them), the labels are what
%  the reader sees. See CLAUDE.md -> "Language".
% =============================================================================

% ---- Theorem environments (shared counter, numbered per section) ------------
% amsthm is already loaded in main.tex.
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[section]
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}
\newtheorem{proposition}[theorem]{Proposition}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}
\newtheorem{example}[theorem]{Example}
\newtheorem{algorithm}[theorem]{Algorithm}

\theoremstyle{remark}
\newtheorem{remark}[theorem]{Remark}
\newtheorem*{remark*}{Remark}

% ---- Number sets ------------------------------------------------------------
% \providecommand so this addon can be merged into a preamble that already
% defines some of these (e.g. \R, \C) without a "command already defined" abort.
% (In the current standalone main.tex nothing pre-exists, so these behave exactly
%  like \newcommand — zero change here, safety only on later merge.)
\providecommand{\R}{\mathbb{R}}
\providecommand{\N}{\mathbb{N}}
\providecommand{\Z}{\mathbb{Z}}
\providecommand{\Q}{\mathbb{Q}}
\providecommand{\C}{\mathbb{C}}
\providecommand{\indicator}{\mathbbm{1}}   % requires \usepackage{bbm} (no \mathbb{1} fallback: amsfonts blackboard has no digits)

% ---- Probability & statistics ----------------------------------------------
% \ifdefined-guarded so merging into an existing preamble never aborts with
% "command already defined" (mirrors the guards on \rank/\diag/\sgn/\supp below).
\ifdefined\E\else\DeclareMathOperator{\E}{\mathbb{E}}\fi          % expectation
\ifdefined\Var\else\DeclareMathOperator{\Var}{Var}\fi
\ifdefined\Cov\else\DeclareMathOperator{\Cov}{Cov}\fi
\ifdefined\Corr\else\DeclareMathOperator{\Corr}{Corr}\fi
\ifdefined\Bias\else\DeclareMathOperator{\Bias}{Bias}\fi
\ifdefined\MSE\else\DeclareMathOperator{\MSE}{MSE}\fi
\ifdefined\Prob\else\DeclareMathOperator{\Prob}{\mathbb{P}}\fi    % \Pr is LaTeX-internal -> \Prob
\ifdefined\argmax\else\DeclareMathOperator*{\argmax}{arg\,max}\fi
\ifdefined\argmin\else\DeclareMathOperator*{\argmin}{arg\,min}\fi
% Guarded: `physics` already defines \rank; guard the rest so the addon works
% whether or not physics is loaded.
\ifdefined\rank\else\DeclareMathOperator{\rank}{rank}\fi
\ifdefined\diag\else\DeclareMathOperator{\diag}{diag}\fi
\ifdefined\sgn\else \DeclareMathOperator{\sgn}{sgn}\fi
\ifdefined\supp\else\DeclareMathOperator{\supp}{supp}\fi

% conditional probability / "given": \Prob(A \given B).
% Robust with plain parens. For a stretchy bar use \left(\,\cdot\,\middle|\,\cdot\,\right).
\providecommand{\given}{\mid}
% independence  X \indep Y
\providecommand{\indep}{\perp\!\!\!\perp}
% i.i.d.  X_1,\dots,X_n \iid \Normal(0,1)
\providecommand{\iid}{\overset{\text{iid}}{\sim}}
% convergence in distribution / probability
\providecommand{\convd}{\overset{d}{\to}}
\providecommand{\convp}{\overset{p}{\to}}

% ---- Distributions ----------------------------------------------------------
\providecommand{\Normal}{\mathcal{N}}
\providecommand{\Bernoulli}{\mathrm{Bernoulli}}
\providecommand{\Binomial}{\mathrm{Bin}}
\providecommand{\Poisson}{\mathrm{Poisson}}
\providecommand{\Uniform}{\mathrm{Unif}}
\providecommand{\Expo}{\mathrm{Exp}}

% ---- Vectors ----------------------------------------------------------------
% ONE macro, ONE place to switch. \vek{x}, \vek{\beta}, \vek{1} (all-ones vector).
%
% The rendering is deliberately centralized here: a lecturer typically writes
% vectors underlined (or arrowed) on the board, which is unusual in a typeset
% script — the standard would be bold. With the macro that question is a
% one-liner, not a migration: in the course this tool was built on, the notation
% previously hung on ~390 individual \underline{} across 14,500 lines.
%
% To switch to the board convention: swap the two lines. \boldsymbol (not
% \mathbf!) is the right choice, because \mathbf does NOT set Greek letters
% bold — \vek{\beta} and \vek{\mu} will occur.
% \providecommand{\vek}[1]{\underline{#1}}   % board convention (as in the video)
\providecommand{\vek}[1]{\boldsymbol{#1}}     % book convention (default)

% ---- Estimator / transpose --------------------------------------------------
\providecommand{\est}[1]{\hat{#1}}            % \est{\theta} -> \hat\theta
% \T begins with ^ : always group exponents before it, otherwise "Double superscript"
% (i.e. (X^{-1})\T instead of X^{-1}\T). Usable only in math mode.
\providecommand{\T}{^{\!\top}}                % A\T  (transpose)
